    function loadFile (i) {


        $(".documentTitle").text(i.replace(/%20/gi, " "));

        $(".inAndOut").load("../e/" + i + ".txt", function (file, status) {

            if (status == "error") {

                $(".inAndOut").text("새로운 문서를 작성합니다.");
            }

            history.pushState("", "", i);
        });
    }

    var u = decodeURI(document.URL.split("/").pop());

    if(u != ""){

        loadFile(u.replace(/ /gi, "%20"));

    } else {

        loadFile("Lorem%20ipsum");
    }
